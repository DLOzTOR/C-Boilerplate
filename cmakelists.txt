cmake_minimum_required(VERSION 3.16)
project(BoilerplateCProject)

# Specify C standard
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)

# Add source headers
set(SRC_DIR "${CMAKE_SOURCE_DIR}/src")
set(RESOURCE_DIR "${CMAKE_SOURCE_DIR}/resources")
set(DEST_DIR "${CMAKE_BINARY_DIR}/resources")

# Add target and copy resources only when needed
file(GLOB RESOURCE_FILES "${RESOURCE_DIR}/*")
add_custom_command(
  OUTPUT ${RESOURCE_FILES}
  COMMAND ${CMAKE_COMMAND} -E copy_directory ${RESOURCE_DIR} ${DEST_DIR}
  DEPENDS ${RESOURCE_FILES}
  COMMENT "Copying updated resources to build folder"
)
add_custom_target(copy_resources ALL DEPENDS ${RESOURCE_FILES})

# Add executable
add_executable(program ${SRC_DIR}/main.c)

# Add include directories to the executable
target_include_directories(program PRIVATE ${SRC_DIR})
